  426    *                                                                         *

  429    *                                                                         *
  430    * PARAMETERS :  -Module: FLEXCAN_A, FLEXCAN_B or FLEXCAN_C                *
  431    *               -BufNum: buffer number for transmit                       *
  432    *               -IdType: 0 means standard format;1 means entend format    *
  433    *               -Id: the message ID to be transmitted                     *
  434    *               -DataLength: data length of the frame data                *
  435    *               -Tx_Buffer: the data array to be transmitted              *
  436    *                                                                         *
  437    * RETURN :      -result: CAN_BUSY or CAN_OK.                              *
  438    ***************************************************************************/
  439   CAN_Return ec_can_transmit(uint8 Module,uint8 BufNum,uint8 IdType,uint32 Id,
  440     uint8 DataLength,uint8 Tx_Buffer[])
  441   {
  442     uint8 i;
  443     CAN_Return result = CAN_BUSY;
  444     if (BufNum<8)
  445       BufNum = 8;
  446     else if ((BufNum>15)&(BufNum<56))
  447       BufNum = 15;
  448     else if (BufNum>63)
  449       BufNum = 63;
  450     else {
  451       if (FLEXCAN(Module).BUF[BufNum].CS.B.CODE == TX_MB_CODE_TRANSMIT) {
  452         result = CAN_BUSY;
  453       } else {
  454         if (IdType == 0) {
  455           FLEXCAN(Module).BUF[BufNum].CS.B.IDE = MB_IDType_ST;
  456           FLEXCAN(Module).BUF[BufNum].ID.B.STD_ID = Id;
  457           FLEXCAN(Module).BUF[BufNum].CS.B.SRR = 0;
  458         } else {
  459           FLEXCAN(Module).BUF[BufNum].CS.B.IDE = MB_IDType_EX;
  460           FLEXCAN(Module).BUF[BufNum].ID.R = (uint32)(Id&0x1FFFFFFF);
  461           FLEXCAN(Module).BUF[BufNum].CS.B.SRR = 1;
  462         }
  463   
  464         FLEXCAN(Module).BUF[BufNum].CS.B.RTR = 0;
  465         for (i=0; i<DataLength; i++) {
  466           FLEXCAN(Module).BUF[BufNum].DATA.B[i] = Tx_Buffer[i];
  467         }
  468   
  469         FLEXCAN(Module).BUF[BufNum].CS.B.LENGTH = DataLength;
  470         FLEXCAN(Module).BUF[BufNum].CS.B.CODE = TX_MB_CODE_TRANSMIT;
  471         result = CAN_OK;
  472       }
  473     }
  474   
  475     return(result);
  476   }
